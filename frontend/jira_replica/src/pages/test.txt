import { useState, useEffect } from 'react';
import { useParams, useNavigate } from 'react-router-dom';
import axios from '../api/axios';
import Navbar from '../components/Navbar';
import './Issue.css';

const PRIORITY_OPTIONS = ['High', 'Medium', 'Low'];
const STATUS_OPTIONS = ['To Do', 'In Progress', 'Done'];

const PRIORITY_CLASSES = {
  High: 'issue__badge--high',
  Medium: 'issue__badge--medium',
  Low: 'issue__badge--low',
};

const Issue = () => {
  const { id } = useParams();
  const navigate = useNavigate();

  const [issue, setIssue] = useState(null);
  const [users, setUsers] = useState([]);
  const [loading, setLoading] = useState(true);
  const [editing, setEditing] = useState(false);
  const [form, setForm] = useState({});
  const [saveLoading, setSaveLoading] = useState(false);
  const [error, setError] = useState(null);
  const [showDeleteConfirm, setShowDeleteConfirm] = useState(false);

  const fetchData = async () => {
    try {
      const [issueRes, usersRes] = await Promise.all([
        axios.get(`/issues/get/${id}`),
        axios.get('/users/getAll'),
      ]);
      setIssue(issueRes.data);
      setUsers(usersRes.data);
      setForm({
        title: issueRes.data.title,
        description: issueRes.data.description || '',
        status: issueRes.data.status,
        priority: issueRes.data.priority,
        assignee_id: issueRes.data.assignee_id || '',
      });
    } catch (err) {
      setError('Erreur lors du chargement.');
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchData();
  }, [id]);

  const handleChange = (e) => {
    setForm({ ...form, [e.target.name]: e.target.value });
  };

  const handleSave = async () => {
    setSaveLoading(true);
    setError(null);
    try {
      const res = await axios.put(`/issues/update/${id}`, {
        updates: {
          title: form.title,
          description: form.description,
          status: form.status,
          priority: form.priority,
          assignee_id: form.assignee_id || null,
        },
      });
      setIssue(res.data.issue);
      setEditing(false);
    } catch (err) {
      setError('Erreur lors de la sauvegarde.');
    } finally {
      setSaveLoading(false);
    }
  };

  const handleDelete = async () => {
    try {
      await axios.delete(`/issues/delete/${id}`);
      navigate(`/projects/${issue.project_id}`);
    } catch (err) {
      setError('Erreur lors de la suppression.');
    }
  };

  const formatDate = (dateStr) => {
    return new Date(dateStr).toLocaleDateString('fr-FR', {
      day: 'numeric',
      month: 'long',
      year: 'numeric',
      hour: '2-digit',
      minute: '2-digit',
    });
  };

  const getAssigneEmail = () => {
    if (!issue?.assignee_id) return 'Non assigné';
    return users.find(u => u.id === issue.assignee_id)?.email || 'Non assigné';
  };

  if (loading) return (
    <div className="issue-page">
      <Navbar />
      <p className="issue-page__loading">Chargement...</p>
    </div>
  );

  if (error && !issue) return (
    <div className="issue-page">
      <Navbar />
      <p className="issue-page__error">{error}</p>
    </div>
  );

  return (
    <div className="issue-page">
      <Navbar />

      <main className="issue-page__main">
        {/* Breadcrumb */}
        <div className="issue-page__breadcrumb">
          <span
            className="issue-page__breadcrumb-link"
            onClick={() => navigate(`/projects/${issue.project_id}`)}
          >
            ← Retour au projet
          </span>
        </div>

        <div className="issue-page__content">
          {/* Colonne principale */}
          <div className="issue-page__left">
            {editing ? (
              <input
                className="issue-page__title-input"
                name="title"
                value={form.title}
                onChange={handleChange}
              />
            ) : (
              <h1 className="issue-page__title">{issue.title}</h1>
            )}

            <div className="issue-page__section">
              <h3 className="issue-page__section-title">Description</h3>
              {editing ? (
                <textarea
                  className="issue-page__textarea"
                  name="description"
                  value={form.description}
                  onChange={handleChange}
                  rows={6}
                  placeholder="Ajouter une description..."
                />
              ) : (
                <p className="issue-page__desc">
                  {issue.description || <span className="issue-page__empty">Aucune description</span>}
                </p>
              )}
            </div>

            {error && <p className="issue-page__error-inline">{error}</p>}

            <div className="issue-page__actions">
              {editing ? (
                <>
                  <button
                    className="issue-page__btn issue-page__btn--save"
                    onClick={handleSave}
                    disabled={saveLoading}
                  >
                    {saveLoading ? 'Sauvegarde...' : 'Sauvegarder'}
                  </button>
                  <button
                    className="issue-page__btn issue-page__btn--cancel"
                    onClick={() => { setEditing(false); setError(null); }}
                  >
                    Annuler
                  </button>
                </>
              ) : (
                <>
                  <button
                    className="issue-page__btn issue-page__btn--edit"
                    onClick={() => setEditing(true)}
                  >
                    Modifier
                  </button>
                  <button
                    className="issue-page__btn issue-page__btn--delete"
                    onClick={() => setShowDeleteConfirm(true)}
                  >
                    Supprimer
                  </button>
                </>
              )}
            </div>
          </div>

          {/* Colonne détails */}
          <div className="issue-page__right">
            <div className="issue-page__details">

              <div className="issue-page__detail">
                <span className="issue-page__detail-label">Statut</span>
                {editing ? (
                  <select
                    className="issue-page__select"
                    name="status"
                    value={form.status}
                    onChange={handleChange}
                  >
                    {STATUS_OPTIONS.map(s => (
                      <option key={s} value={s}>{s}</option>
                    ))}
                  </select>
                ) : (
                  <span className="issue-page__detail-value">{issue.status}</span>
                )}
              </div>

              <div className="issue-page__detail">
                <span className="issue-page__detail-label">Priorité</span>
                {editing ? (
                  <select
                    className="issue-page__select"
                    name="priority"
                    value={form.priority}
                    onChange={handleChange}
                  >
                    {PRIORITY_OPTIONS.map(p => (
                      <option key={p} value={p}>{p}</option>
                    ))}
                  </select>
                ) : (
                  <span className={`issue-page__badge ${PRIORITY_CLASSES[issue.priority]}`}>
                    {issue.priority}
                  </span>
                )}
              </div>

              <div className="issue-page__detail">
                <span className="issue-page__detail-label">Assigné à</span>
                {editing ? (
                  <select
                    className="issue-page__select"
                    name="assignee_id"
                    value={form.assignee_id}
                    onChange={handleChange}
                  >
                    <option value="">Non assigné</option>
                    {users.map(u => (
                      <option key={u.id} value={u.id}>{u.email}</option>
                    ))}
                  </select>
                ) : (
                  <span className="issue-page__detail-value">{getAssigneEmail()}</span>
                )}
              </div>

              <div className="issue-page__detail">
                <span className="issue-page__detail-label">Créé le</span>
                <span className="issue-page__detail-value">{formatDate(issue.created_at)}</span>
              </div>

              <div className="issue-page__detail">
                <span className="issue-page__detail-label">Modifié le</span>
                <span className="issue-page__detail-value">{formatDate(issue.updated_at)}</span>
              </div>

            </div>
          </div>
        </div>
      </main>

      {/* Confirm delete */}
      {showDeleteConfirm && (
        <div className="issue-page__overlay" onClick={() => setShowDeleteConfirm(false)}>
          <div className="issue-page__modal" onClick={(e) => e.stopPropagation()}>
            <h2 className="issue-page__modal-title">Supprimer cette issue ?</h2>
            <p className="issue-page__modal-text">Cette action est irréversible.</p>
            <div className="issue-page__modal-actions">
              <button
                className="issue-page__btn issue-page__btn--cancel"
                onClick={() => setShowDeleteConfirm(false)}
              >
                Annuler
              </button>
              <button
                className="issue-page__btn issue-page__btn--delete"
                onClick={handleDelete}
              >
                Supprimer
              </button>
            </div>
          </div>
        </div>
      )}
    </div>
  );
};

export default Issue;
